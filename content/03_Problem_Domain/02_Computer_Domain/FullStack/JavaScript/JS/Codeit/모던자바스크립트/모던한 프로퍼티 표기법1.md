## 옵셔널 체이닝 (Optional Chaining)
ECMAScript 2020에 등장한 옵셔널 체이닝
중첩된 객체를 다룰때 우리가 예상한 프로퍼티를 가지고 있지 않을 수 있다.
그래서 `null,undefined`가 아니라는 것을 검증하고 접근해야 에러가 발생하지않는다.
```js
function printCatName(user) {
	console.log(user.cat && user.cat.name);
} // 하나라도 없을시 콘솔에서는 undefined나 null을 출력할것이다.
//이러한 방법은 길어질수록 가독성이 나빠진다. 이럴때 사용하는것이 옵셔널 체이닝

function printCatName(user) {
	console.log(user.cat?.name); //Optional Chaining 연산자 ?
}
// 왼편 프로퍼티 값이 undefined또는 null이 아니라면 그다음 프로퍼티 값을 리턴하고 그렇지 않은 경우에는 undefined를 반환

function printCatName(user) {
  console.log((user.cat === null || user.cat === undefined) ? undefined : user.cat.name);
} //풀어쓰면 이렇다.

```
```js
//활용
function printCatName(user) {
	console.log(user.cat?.name ?? '함께 지내는 고양이가 없습니다');
}

const user2 = {
	name: 'Young'
}

printCatName(user2); //함께지내는...
```

## 구조분해(Destricturing)
```js
const rank=['kim','dae','jun','ilm'];
const [macbook,ipad,airpos] = rank; // 구조분해의 기본

console.log(macbook);
console.log(ipad);
console.log(airpos);
// 인덱스 순서에 따라 할당되기때문에 배열의 길이와는 상관이없다.

const [macbook, ipad, airpos, ...coupon] = rank // 이렇게하면 나머지 요소에 배열로 할당하는게 가능하다 항상 오른쪽에 마지막 변수에만 활용할 수 있다는 점을 주의 REST Parameter와 비슷하다
//만약 배열의 길이가 적다면 나머지는 undefined 가 할당이된다.
//아래처럼 기본값을 설정할수있다.
const [macbook, ipad, airpos, coupon='defaultValue'] = rank

```
교차해서 (교환) 할당이 가능하다
```JS
let macbook = 'youn'
let ipad = 'lim'

[macbook,ipad] = [ipad,macbook];

console.log(macbook); // ilm
console.log(ipad); // youn
```

## 구조분해(객체)
```js
const macbook = {
	title: 'macbook 16inch',
	price: 30000
};

const {title,price} = macbook // 배열에서 []쓸것처럼 객체에는 중괄호로 변수를 감싸쓴다.
// 객체는 프로퍼티 네임으로 분해된다.
//객체에 프로퍼티 네임에 왼쪽 변수선언 이름과 똑같으면 그값이 할당되는 방식이다.
const {title,color='red'} = macbook
// 프로퍼티가 없다면 undefined할당 기본값가능
const {title,...rest} = macbook
// 나머지 남은 부분을 하나의 객체로 모을수있따.
cosnt {title: product, ...rest} = macbook
// 새로운 변수를 선언해줄수도있다.

console.log(title);
console.log(color);
console.log(price);
console.log(product);

//새로운 변수는 'seraial-num' 같이 변수로 활용할수없을때
const {title:product, 'seraial-num':seralNum } = macbook;

const { [propertyName] : product } = macbook;
```
